from __future__ import annotations
import json
import threading
from contextlib import contextmanager
from datetime import date, datetime
from pathlib import Path
from typing import Optional

from sqlalchemy import create_engine, Column, Integer, Float, String, Text, DateTime, Date, Boolean
from sqlalchemy.orm import DeclarativeBase, Session, sessionmaker

from config import settings


class Base(DeclarativeBase):
    pass


class Trade(Base):
    __tablename__ = "trades"
    id = Column(Integer, primary_key=True, autoincrement=True)
    date = Column(Date, nullable=False, index=True)
    symbol = Column(String(32), nullable=False)
    direction = Column(String(8), nullable=False)  # LONG / SHORT
    entry_price = Column(Float, nullable=False)
    exit_price = Column(Float)
    quantity = Column(Integer, nullable=False)
    pnl = Column(Float)
    entry_time = Column(DateTime, nullable=False)
    exit_time = Column(DateTime)
    exit_reason = Column(String(32))
    strategy_params = Column(Text)   # JSON
    commission = Column(Float, default=0.0)
    is_paper = Column(Boolean, default=True)


class DailySummary(Base):
    __tablename__ = "daily_summary"
    id = Column(Integer, primary_key=True, autoincrement=True)
    date = Column(Date, nullable=False, unique=True)
    total_trades = Column(Integer, default=0)
    winning_trades = Column(Integer, default=0)
    losing_trades = Column(Integer, default=0)
    gross_profit = Column(Float, default=0.0)
    gross_loss = Column(Float, default=0.0)
    net_pnl = Column(Float, default=0.0)
    win_rate_pct = Column(Float, default=0.0)
    best_trade_pnl = Column(Float, default=0.0)
    worst_trade_pnl = Column(Float, default=0.0)


class BotLog(Base):
    __tablename__ = "bot_logs"
    id = Column(Integer, primary_key=True, autoincrement=True)
    timestamp = Column(DateTime, default=datetime.utcnow, index=True)
    level = Column(String(16), nullable=False)
    message = Column(Text, nullable=False)


class ConfigHistory(Base):
    __tablename__ = "config_history"
    id = Column(Integer, primary_key=True, autoincrement=True)
    timestamp = Column(DateTime, default=datetime.utcnow)
    param_name = Column(String(64), nullable=False)
    old_value = Column(Text)
    new_value = Column(Text)
    changed_by = Column(String(32), default="dashboard")


class TradeLogger:
    def __init__(self):
        db_path = settings.log_dir / "trades.db"
        self._engine = create_engine(f"sqlite:///{db_path}", echo=False)
        Base.metadata.create_all(self._engine)
        self._Session = sessionmaker(bind=self._engine)
        self._lock = threading.Lock()
    
    @contextmanager
    def _session(self):
        with self._lock:
            s = self._Session()
            try:
                yield s
                s.commit()
            except Exception:
                s.rollback()
                raise
            finally:
                s.close()
    
    def log_trade_entry(self, symbol, direction, entry_price, quantity, entry_time, strategy_params=None, is_paper=True):
        with self._session() as s:
            trade = Trade(
                date=entry_time.date(),
                symbol=symbol,
                direction=direction,
                entry_price=entry_price,
                quantity=quantity,
                entry_time=entry_time,
                strategy_params=json.dumps(strategy_params or {}),
                is_paper=is_paper,
            )
            s.add(trade)
            s.flush()
            return trade.id
    
    def log_trade_exit(self, trade_id, exit_price, exit_time, pnl, exit_reason, commission=0.0):
        with self._session() as s:
            trade = s.get(Trade, trade_id)
            if trade:
                trade.exit_price = exit_price
                trade.exit_time = exit_time
                trade.pnl = pnl
                trade.exit_reason = exit_reason
                trade.commission = commission
    
    def log_trade(self, trade_dict):
        """Log a completed trade from backtest or order manager."""
        with self._session() as s:
            entry_time = trade_dict.get("entry_time")
            if isinstance(entry_time, str):
                entry_time = datetime.fromisoformat(entry_time)
            exit_time = trade_dict.get("exit_time")
            if isinstance(exit_time, str):
                exit_time = datetime.fromisoformat(exit_time) if exit_time else None
            
            trade = Trade(
                date=entry_time.date() if entry_time else date.today(),
                symbol=trade_dict.get("symbol", "UNKNOWN"),
                direction=trade_dict.get("direction", "LONG"),
                entry_price=trade_dict.get("entry_price", 0.0),
                exit_price=trade_dict.get("exit_price"),
                quantity=trade_dict.get("quantity", 0),
                pnl=trade_dict.get("pnl"),
                entry_time=entry_time,
                exit_time=exit_time,
                exit_reason=trade_dict.get("exit_reason"),
                strategy_params=json.dumps(trade_dict.get("strategy_params", {})),
                commission=trade_dict.get("commission", 0.0),
                is_paper=trade_dict.get("is_paper", True),
            )
            s.add(trade)
    
    def log_config_change(self, param_name, old_value, new_value, changed_by="dashboard"):
        with self._session() as s:
            entry = ConfigHistory(
                param_name=param_name,
                old_value=str(old_value),
                new_value=str(new_value),
                changed_by=changed_by,
            )
            s.add(entry)
    
    def log_bot_event(self, level, message):
        with self._session() as s:
            log = BotLog(level=level, message=message, timestamp=datetime.utcnow())
            s.add(log)
    
    def get_trades(self, from_date=None, to_date=None, symbol=None, limit=200):
        with self._session() as s:
            q = s.query(Trade)
            if from_date:
                q = q.filter(Trade.date >= from_date)
            if to_date:
                q = q.filter(Trade.date <= to_date)
            if symbol:
                q = q.filter(Trade.symbol == symbol)
            trades = q.order_by(Trade.entry_time.desc()).limit(limit).all()
            return [self._trade_to_dict(t) for t in trades]
    
    def get_today_summary(self):
        today = date.today()
        with self._session() as s:
            trades = s.query(Trade).filter(
                Trade.date == today,
                Trade.exit_time.isnot(None)
            ).all()
            
            total = len(trades)
            wins = sum(1 for t in trades if t.pnl and t.pnl > 0)
            losses = total - wins
            net_pnl = sum(t.pnl for t in trades if t.pnl) 
            return {
                "date": str(today),
                "total_trades": total,
                "winning_trades": wins,
                "losing_trades": losses,
                "net_pnl": round(net_pnl, 2),
                "win_rate": round(wins / total * 100, 1) if total else 0,
            }
    
    def get_daily_pnl(self, days=30):
        from sqlalchemy import func
        with self._session() as s:
            rows = (
                s.query(Trade.date, func.sum(Trade.pnl).label("pnl"), func.count(Trade.id).label("trades"))
                .filter(Trade.exit_time.isnot(None))
                .group_by(Trade.date)
                .order_by(Trade.date.desc())
                .limit(days)
                .all()
            )
            return [{"date": str(r.date), "pnl": round(r.pnl or 0, 2), "trades": r.trades} for r in reversed(rows)]
    
    def get_equity_curve(self, limit=1000):
        with self._session() as s:
            trades = (
                s.query(Trade)
                .filter(Trade.exit_time.isnot(None), Trade.pnl.isnot(None))
                .order_by(Trade.exit_time)
                .limit(limit)
                .all()
            )
            equity = settings.trading_capital
            points = []
            for t in trades:
                equity += t.pnl
                points.append({"time": t.exit_time.isoformat(), "equity": round(equity, 2)})
            return points
    
    def get_recent_logs(self, level=None, limit=100):
        with self._session() as s:
            q = s.query(BotLog)
            if level:
                q = q.filter(BotLog.level == level.upper())
            logs = q.order_by(BotLog.timestamp.desc()).limit(limit).all()
            return [{"time": l.timestamp.isoformat(), "level": l.level, "message": l.message} for l in reversed(logs)]
    
    def _trade_to_dict(self, t):
        return {
            "id": t.id,
            "date": str(t.date),
            "symbol": t.symbol,
            "direction": t.direction,
            "entry_price": t.entry_price,
            "exit_price": t.exit_price,
            "quantity": t.quantity,
            "pnl": round(t.pnl, 2) if t.pnl is not None else None,
            "entry_time": t.entry_time.isoformat() if t.entry_time else None,
            "exit_time": t.exit_time.isoformat() if t.exit_time else None,
            "exit_reason": t.exit_reason,
            "commission": t.commission,
            "is_paper": t.is_paper,
        }

import asyncio
import threading

class TelegramNotifier:
    def __init__(self, settings, order_manager=None, risk_manager=None):
        self._token = settings.telegram_bot_token
        self._chat_id = settings.telegram_chat_id
        self._order_manager = order_manager
        self._loop = None
        self._app = None
        
    def start(self):
        self._thread = threading.Thread(target=self._run_bot, daemon=True)
        self._thread.start()

class TradeLogger:
    """
    SQLite-backed trade journal.

    Tables
    ------
    trades          Round-trip trade log (entry + exit in one row).
    daily_summary   Computed EOD stats — written by compute_daily_summary().
    bot_logs        Structured log stream (INFO/WARNING/ERROR).
    config_history  Before/after record of every parameter change.

    Usage
    -----
        tl = TradeLogger()
        tl.log_trade({"type": "ENTRY", ...})   # called by OrderManager
        tl.log_trade({"type": "EXIT", ...})     # called by OrderManager
        tl.get_today_summary()                  # called by dashboard
    """

    def __init__(self) -> None:
        from config import settings
        db_path = settings.log_dir / "trades.db"
        self._engine = create_engine(
            f"sqlite:///{db_path}", echo=False, connect_args={"check_same_thread": False}
        )
        _Base.metadata.create_all(self._engine)
        self._Session = sessionmaker(bind=self._engine, expire_on_commit=False)
        self._lock = threading.Lock()
        # symbol → open trade id (used to match EXIT to ENTRY row)
        self._open_ids: dict[str, int] = {}

    # ── Session context manager ────────────────────────────────────────────────

    @contextmanager
    def _session(self) -> Iterator[Session]:
        with self._lock:
            s = self._Session()
            try:
                yield s
                s.commit()
            except Exception:
                s.rollback()
                raise
            finally:
                s.close()

    # ── Trade logging ──────────────────────────────────────────────────────────

    def log_trade(self, data: dict) -> None:
        """
        Called by OrderManager with type=ENTRY | EXIT | PARTIAL_EXIT.

        ENTRY  → inserts a new row; stores id in _open_ids[symbol].
        EXIT   → updates the open row with exit price + P&L.
        """
        trade_type = data.get("type", "")

        if trade_type == "ENTRY":
            ts_str = data.get("timestamp", "")
            try:
                entry_time = datetime.fromisoformat(ts_str) if ts_str else datetime.now()
            except ValueError:
                entry_time = datetime.now()

            with self._session() as s:
                trade = _Trade(
                    date=entry_time.date(),
                    symbol=data.get("symbol", ""),
                    direction=data.get("direction", "LONG"),
                    entry_price=data.get("price", 0.0),
                    quantity=data.get("quantity", 0),
                    initial_sl=data.get("sl"),
                    target=data.get("target"),
                    entry_time=entry_time,
                    strategy_params=json.dumps({
                        "confidence": data.get("confidence"),
                        "broker_symbol": data.get("broker_symbol"),
                    }),
                    is_paper=data.get("mode", "PAPER") == "PAPER",
                )
                s.add(trade)
                s.flush()
                self._open_ids[data.get("symbol", "")] = trade.id
                logger.debug("TradeLogger: ENTRY logged id={}", trade.id)

        elif trade_type == "EXIT":
            ts_str = data.get("timestamp", "")
            try:
                exit_time = datetime.fromisoformat(ts_str) if ts_str else datetime.now()
            except ValueError:
                exit_time = datetime.now()

            symbol = data.get("symbol", "")
            trade_id = self._open_ids.pop(symbol, None)

            with self._session() as s:
                if trade_id:
                    trade = s.get(_Trade, trade_id)
                else:
                    # Fallback: find the most recent open trade for this symbol
                    trade = (
                        s.query(_Trade)
                        .filter(_Trade.symbol == symbol, _Trade.exit_time.is_(None))
                        .order_by(_Trade.entry_time.desc())
                        .first()
                    )
                if trade:
                    trade.exit_price = data.get("price")
                    trade.exit_time  = exit_time
                    trade.pnl        = data.get("pnl")
                    trade.exit_reason= data.get("reason", "UNKNOWN")
                    logger.debug("TradeLogger: EXIT logged id={} pnl={}", trade.id, trade.pnl)
                else:
                    logger.warning("TradeLogger: no open trade found for {}", symbol)

        # PARTIAL_EXIT — just log a note; main row stays open
        elif trade_type == "PARTIAL_EXIT":
            logger.debug(
                "TradeLogger: partial exit {symbol} qty={qty} pnl={pnl}",
                symbol=data.get("symbol"),
                qty=data.get("partial_qty"),
                pnl=data.get("partial_pnl"),
            )

    # ── Query helpers ──────────────────────────────────────────────────────────

    def get_trades(
        self,
        from_date: Optional[date] = None,
        to_date: Optional[date] = None,
        symbol: Optional[str] = None,
        direction: Optional[str] = None,
        limit: int = 200,
    ) -> list[dict]:
        with self._session() as s:
            q = s.query(_Trade).filter(_Trade.exit_time.isnot(None))
            if from_date:
                q = q.filter(_Trade.date >= from_date)
            if to_date:
                q = q.filter(_Trade.date <= to_date)
            if symbol:
                q = q.filter(_Trade.symbol == symbol)
            if direction:
                q = q.filter(_Trade.direction == direction.upper())
            rows = q.order_by(_Trade.entry_time.desc()).limit(limit).all()
            return [_trade_to_dict(r) for r in rows]

    def get_open_trades(self) -> list[dict]:
        with self._session() as s:
            rows = s.query(_Trade).filter(_Trade.exit_time.is_(None)).all()
            return [_trade_to_dict(r) for r in rows]

    def get_today_summary(self) -> dict:
        today = date.today()
        with self._session() as s:
            trades = (
                s.query(_Trade)
                .filter(_Trade.date == today, _Trade.exit_time.isnot(None))
                .all()
            )
        total  = len(trades)
        wins   = sum(1 for t in trades if t.pnl and t.pnl > 0)
        losses = total - wins
        net    = round(sum(t.pnl for t in trades if t.pnl is not None), 2)
        return {
            "date": str(today),
            "total_trades": total,
            "winning_trades": wins,
            "losing_trades": losses,
            "net_pnl": net,
            "win_rate": round(wins / total * 100, 1) if total else 0.0,
        }

    def get_daily_pnl(self, days: int = 30) -> list[dict]:
        with self._session() as s:
            rows = (
                s.query(
                    _Trade.date,
                    func.sum(_Trade.pnl).label("pnl"),
                    func.count(_Trade.id).label("trades"),
                )
                .filter(_Trade.exit_time.isnot(None))
                .group_by(_Trade.date)
                .order_by(_Trade.date.desc())
                .limit(days)
                .all()
            )
        return [
            {"date": str(r.date), "pnl": round(r.pnl or 0.0, 2), "trades": r.trades}
            for r in reversed(rows)
        ]

    def get_equity_curve(self, limit: int = 2000) -> list[dict]:
        from config import settings
        with self._session() as s:
            rows = (
                s.query(_Trade)
                .filter(_Trade.exit_time.isnot(None), _Trade.pnl.isnot(None))
                .order_by(_Trade.exit_time)
                .limit(limit)
                .all()
            )
        equity = settings.trading_capital
        pts: list[dict] = []
        for r in rows:
            equity += r.pnl
            pts.append({"time": r.exit_time.isoformat(), "equity": round(equity, 2)})
        return pts

    def get_recent_logs(self, level: Optional[str] = None, limit: int = 100) -> list[dict]:
        with self._session() as s:
            q = s.query(_BotLog)
            if level and level.upper() != "ALL":
                q = q.filter(_BotLog.level == level.upper())
            rows = q.order_by(_BotLog.timestamp.desc()).limit(limit).all()
        return [
            {"time": r.timestamp.isoformat(), "level": r.level, "message": r.message}
            for r in reversed(rows)
        ]

    def log_bot_event(self, level: str, message: str) -> None:
        with self._session() as s:
            s.add(_BotLog(level=level.upper(), message=message, timestamp=datetime.utcnow()))

    def log_config_change(
        self,
        param_name: str,
        old_value: Any,
        new_value: Any,
        changed_by: str = "dashboard",
    ) -> None:
        with self._session() as s:
            s.add(_ConfigHistory(
                param_name=param_name,
                old_value=str(old_value),
                new_value=str(new_value),
                changed_by=changed_by,
                timestamp=datetime.utcnow(),
            ))

    def compute_daily_summary(self, target_date: Optional[date] = None) -> None:
        """Compute and upsert the daily summary row (call at EOD)."""
        d = target_date or date.today()
        with self._session() as s:
            trades = (
                s.query(_Trade)
                .filter(_Trade.date == d, _Trade.exit_time.isnot(None))
                .all()
            )
            if not trades:
                return

            total  = len(trades)
            wins   = [t for t in trades if t.pnl and t.pnl > 0]
            losses = [t for t in trades if not t.pnl or t.pnl <= 0]
            net    = sum(t.pnl for t in trades if t.pnl)
            gross_p = sum(t.pnl for t in wins if t.pnl)
            gross_l = abs(sum(t.pnl for t in losses if t.pnl))
            pnls   = [t.pnl for t in trades if t.pnl]

            row = s.query(_DailySummary).filter(_DailySummary.date == d).first()
            if row is None:
                row = _DailySummary(date=d)
                s.add(row)

            row.total_trades    = total
            row.winning_trades  = len(wins)
            row.losing_trades   = len(losses)
            row.gross_profit    = round(gross_p, 2)
            row.gross_loss      = round(gross_l, 2)
            row.net_pnl         = round(net, 2)
            row.win_rate_pct    = round(len(wins) / total * 100, 1) if total else 0.0
            row.best_trade_pnl  = round(max(pnls), 2)
            row.worst_trade_pnl = round(min(pnls), 2)

from telegram import Bot, Update

class TelegramNotifier:
    def __init__(self, settings):
        self._token = settings.telegram_bot_token
        self._chat_id = settings.telegram_chat_id
        self._loop: asyncio.AbstractEventLoop | None = None
        self._app: Application | None = None
        self._ctx = {}  # order_manager, risk_manager etc
    
    def set_context(self, **kwargs):
        self._ctx.update(kwargs)
    
    def start(self):
        t = threading.Thread(target=self._run_forever, daemon=True)
        t.start()
    
    def _run_forever(self):
        self._loop = asyncio.new_event_loop()
        asyncio.set_event_loop(self._loop)
        try:
            self._loop.run_until_complete(self._start_polling())
        except Exception as exc:
            logger.error("Telegram bot error: {}", exc)
    
    async def _start_polling(self):
        self._app = Application.builder().token(self._token).build()
        
        self._app.add_handler(CommandHandler("start", self._cmd_start))
        self._app.add_handler(CommandHandler("status", self._cmd_status))
        self._app.add_handler(CommandHandler("today", self._cmd_today))
        self._app.add_handler(CommandHandler("pause", self._cmd_pause))
        self._app.add_handler(CommandHandler("resume", self._cmd_resume))
        self._app.add_handler(CommandHandler("squareoff", self._cmd_squareoff))
        
        await self._app.initialize()
        await self._app.start()
        await self._app.updater.start_polling(drop_pending_updates=True)
        await asyncio.Event().wait()  # run forever
    
    def send(self, message: str) -> None:
        if self._loop is None or self._app is None:
            logger.debug("[Telegram not started] {}", message)
            return
        fut = asyncio.run_coroutine_threadsafe(
            self._app.bot.send_message(
                chat_id=self._chat_id, text=message, parse_mode="HTML"
            ),
            self._loop
        )
        try:

        except Exception as exc:
            logger.warning("Telegram send failed: {}", exc)
